#version 400 core

in vec3 v_position;
in vec2 v_textureCoords;

out vec2 f_textureCoord;

uniform mat4 transformationMatrix;  //do ustalania pozycji każdego z obiektów
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;
void main(void)
{
    gl_Position =  projectionMatrix * viewMatrix * transformationMatrix*vec4(v_position, 1.0);
    f_textureCoord = v_textureCoords;
}
